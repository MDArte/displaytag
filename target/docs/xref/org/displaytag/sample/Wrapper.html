12  package org.displaytag.sample;
13  
14  import java.text.DecimalFormat;
15  
16  import org.apache.commons.lang.time.FastDateFormat;
17  import org.displaytag.decorator.TableDecorator;
18  
19  
20  /***
21   * This class is a decorator of the TestObjects that we keep in our List. This class provides a number of methods for
22   * formatting data, creating dynamic links, and exercising some aspects of the display:table API functionality.
23   * @author epesh
24   * @author Fabrizio Giustina
25   * @version $Revision: 1.12 $ ($Author: fgiust $)
26   */
27  public class Wrapper extends TableDecorator
28  {
29  
30      /***
31       * FastDateFormat used to format dates in getDate().
32       */
33      private FastDateFormat dateFormat;
34  
35      /***
36       * DecimalFormat used to format money in getMoney().
37       */
38      private DecimalFormat moneyFormat;
39  
40      /***
41       * Creates a new Wrapper decorator who's job is to reformat some of the data located in our TestObject's.
42       */
43      public Wrapper()
44      {
45          super();
46  
47          // Formats for displaying dates and money.
48  
49          this.dateFormat = FastDateFormat.getInstance("MM/dd/yy"); //$NON-NLS-1$
50          this.moneyFormat = new DecimalFormat("$ #,###,###.00"); //$NON-NLS-1$
51      }
52  
53      /***
54       * Test method which always returns a null value.
55       * @return <code>null</code>
56       */
57      public String getNullValue()
58      {
59          return null;
60      }
61  
62      /***
63       * Returns the date as a String in MM/dd/yy format.
64       * @return formatted date
65       */
66      public String getDate()
67      {
68          return this.dateFormat.format(((ListObject) this.getCurrentRowObject()).getDate());
69      }
70  
71      /***
72       * Returns the money as a String in $ #,###,###.00 format.
73       * @return String
74       */
75      public String getMoney()
76      {
77          return this.moneyFormat.format(((ListObject) this.getCurrentRowObject()).getMoney());
78      }
79  
80      /***
81       * Returns the TestObject's ID as a hyperlink that the person can click on and "drill down" for more details.
82       * @return String
83       */
84      public String getLink1()
85      {
86          ListObject object = (ListObject) getCurrentRowObject();
87          int index = getListIndex();
88  
89          return "<a href=\"details.jsp?index=" //$NON-NLS-1$
90              + index + "\">" //$NON-NLS-1$
91              + object.getId() + "</a>"; //$NON-NLS-1$
92      }
93  
94      /***
95       * Returns an "action bar" of sorts that allow the user to perform various actions on the TestObject based on it's
96       * id.
97       * @return String
98       */
99      public String getLink2()
100     {
101         ListObject object = (ListObject) getCurrentRowObject();
102         int id = object.getId();
103 
104         return "<a href=\"details.jsp?id=" //$NON-NLS-1$
105             + id + "&action=view\">View</a> | " //$NON-NLS-1$
106             + "<a href=\"details.jsp?id=" //$NON-NLS-1$
107             + id + "&action=edit\">Edit</a> | " //$NON-NLS-1$
108             + "<a href=\"details.jsp?id=" //$NON-NLS-1$
109             + id + "&action=delete\">Delete</a>"; //$NON-NLS-1$
110     }
111 }
